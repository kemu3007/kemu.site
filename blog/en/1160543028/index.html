Automatic generation of Django Admin

発行日時: 2022.03.06 07:45

最終更新日: 2022.03.06 03:21

I forgot to register admin and it was a hassle, so I'll leave it as a note.

Convenient timing

  • I don't want to put too much effort into Django Admin, but I want the bare minimum of functionality.
  • I want to do something about forgetting admin.site.register when adding models.

from typing import Type
from django.contrib.admin.options import ModelAdmin
from django.contrib.admin.sites import AdminSite
from django.apps import apps
from django.db import models
from colorfield.fields import ColorField


def get_admin_model(model: Type[models.Model]) -> Type[ModelAdmin]:
    list_display_fields = [models.CharField, models.IntegerField, models.PositiveIntegerField, models.ForeignKey, models.BooleanField, ColorField]
    list_editable_field = [models.IntegerField, models.PositiveIntegerField, models.BooleanField, ColorField]
    list_display = ["pk"]
    list_editable = []
    for field in model._meta.fields:
        if type(field) in list_display_fields:
            list_display.append(field.name)
        if type(field) in list_editable_field and field.editable:
            list_editable.append(field.name)
    return type(f"{model._meta.model_name}ModelAdmin", (ModelAdmin, ), {"list_display": list_display, "list_editable": list_editable, "ordering": model._meta.ordering or ["pk"]})


class AutoGenerateAdminSite(AdminSite):
    display_app_list = ['sites', 'shared', 'article', 'contact']

    def __init__(self, *args, **kwargs) -> None:
        super().__init__(*args, **kwargs)
        self._registry = {model: get_admin_model(model)(model=model, admin_site=self) for model in apps.get_models()}

    def _build_app_dict(self, request):
        app_dict = super()._build_app_dict(request)
        return dict(filter(lambda app: app[0] in self.display_app_list, app_dict.items()))

Create an AutoGenerateAdminSite that extends django.contrib.admin.sites.AdminSite and add any model to AutoGenerateAdminSite._registry .

Because it uses django.apps.get_models, it also displays data that even admin does not want to see, such as sessions.

In order to exclude such data on an app- by- app basis, we narrow down the apps to be displayed in the _build_app_dict.
